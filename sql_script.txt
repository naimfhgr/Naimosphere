CREATE TABLE IF NOT EXISTS air_quality (
    id INT AUTO_INCREMENT PRIMARY KEY,
    time DATETIME NOT NULL,
    latitude FLOAT,
    longitude FLOAT,
    european_aqi INT,
    pm10 FLOAT,
    pm2_5 FLOAT,
    ozone FLOAT,
    birch_pollen INT,
    grass_pollen INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY unique_time (time)
);
